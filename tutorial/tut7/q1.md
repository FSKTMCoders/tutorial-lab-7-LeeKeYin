1. Write statements for each of the following
a. Store ten random integers within 0 to 1000 to a text file name integer.txt.

```java
import java.io.*;
import java.util.Random;
public class T7 {

    public static void main(String[] args) throws IOException {
        PrintWriter out = new PrintWriter(new FileOutputStream("integer.txt"));
        Random rand = new Random();
        for(int i=0; i<10; i++) {
            out.println(rand.nextInt(1001));
        }
        out.close();
    }
}
```

b. Read from the text file generated in a. Display all the integer and the largest integer.

```java
import java.io.*;
import java.util.Scanner;
public class T7 {

    public static void main(String[] args) throws IOException {
        Scanner in = new Scanner(new FileInputStream("integer.txt"));
        int max = Integer.MIN_VALUE;
        System.out.println("All integers: ");
        while(in.hasNextInt()) {
            int num = in.nextInt();
            System.out.println(num);
            if(num>max) {
                max=num;
            }
        }
        System.out.println("Largest integer: " + max);
        in.close();
    }
}
```

c. Store ten random integers within 0 to 1000 to a binary file name integer.dat.

```java
import java.io.*;
import java.util.Random;
public class T7 {

    public static void main(String[] args) throws IOException {
        DataOutputStream out = new DataOutputStream(new FileOutputStream("integer.dat"));
        Random rand = new Random();
        for(int i=0; i<10; i++) {
            out.writeInt(rand.nextInt(1001));
        }
        out.close();
    }
}
```

import java.io.*;
public class T7 {

    public static void main(String[] args) throws IOException {
        DataInputStream in = new DataInputStream(new FileInputStream("integer.dat"));
        int sum=0, count=0;
        System.out.println("All integers: ");
        try {
            while(true) {
                int num = in.readInt();
                System.out.println(num);
                sum += num;
                count++;
            }
        }catch (IOException e) {
            
        }
        double average = (double) sum/count;
        System.out.println("Average: " + average);
        in.close();
    }
}
